const TAB_NAME = "檢測分析紀錄表V.1-2026"; const HEADER_ROW_INDEX = 0; const HEADER_SCAN_MAX_ROWS = 6; const COL_DATE_CANDIDATES = ["日期", "填寫日期", "建立日期", "檢測日期", "報告日期", "時間戳記", "Timestamp"]; const COL_NAME_CANDIDATES = ["客戶本名", "姓名", "客戶", "名字", "本名", "客戶姓名"]; const COL_PAID_CANDIDATES = ["已付款", "已對帳", "付款", "是否已付款", "付款狀態"]; const COL_RESCHEDULE_CANDIDATES = ["改期紀錄", "改期記錄", "改期"]; const COL_BOOKING_CANDIDATES = ["預約日期", "預約", "預約時間", "預約日"]; const COL_INTL_CANDIDATES = ["是否為 國際美金", "是否為國際美金", "國際美金", "是否國際美金"]; const REFUND_KEYWORDS = ["退費"]; function defaultRange_() { const end = new Date(); end.setHours(23, 59, 59, 999); const start = new Date(end); start.setMonth(start.getMonth() - 6); start.setHours(0, 0, 0, 0); return { start, end, label: "last_6_months" }; } function doGet(e) { const params = e && e.parameter ? e.parameter : {}; const action = String(params.action || "summary").toLowerCase(); const status = String(params.status || "not_booked").toLowerCase(); const callback = params.callback; try { if (action === "debug") return output_(buildDebug_(), callback); const range = getRangeFromParams_(params); let payload; if (action === "summary") payload = buildSummary_(range); else if (action === "list") payload = buildList_(status, range); else payload = { ok: false, error: "Unknown action: " + action }; return output_(payload, callback); } catch (err) { return output_({ ok: false, error: String(err && err.message ? err.message : err) }, callback); } } function doOptions(e) { return ContentService.createTextOutput("").setMimeType(ContentService.MimeType.TEXT); } function output_(payload, callback) { const json = JSON.stringify(payload); if (callback) { return ContentService.createTextOutput(callback + "(" + json + ")").setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); } function headerResult_(rowIndex, headers) { return { headerRowIndex: rowIndex, headers, idxDate: findHeaderIndex_(headers, COL_DATE_CANDIDATES), idxName: findHeaderIndex_(headers, COL_NAME_CANDIDATES), idxPaid: findHeaderIndex_(headers, COL_PAID_CANDIDATES), idxReschedule: findHeaderIndex_(headers, COL_RESCHEDULE_CANDIDATES), idxBooking: findHeaderIndex_(headers, COL_BOOKING_CANDIDATES), idxIntl: findHeaderIndex_(headers, COL_INTL_CANDIDATES) }; } function resolveHeader_(values) { if (HEADER_ROW_INDEX && HEADER_ROW_INDEX >= 1) { const headers = (values[HEADER_ROW_INDEX - 1] || []).map(h => cleanText_(h)); return headerResult_(HEADER_ROW_INDEX, headers); } let best = null; const scanMax = Math.min(HEADER_SCAN_MAX_ROWS, values.length); for (let r = 0; r < scanMax; r++) { const headers = (values[r] || []).map(h => cleanText_(h)); const idxDate = findHeaderIndex_(headers, COL_DATE_CANDIDATES); const idxPaid = findHeaderIndex_(headers, COL_PAID_CANDIDATES); const idxBooking = findHeaderIndex_(headers, COL_BOOKING_CANDIDATES); const score = (idxDate >= 0 ? 1 : 0) + (idxPaid >= 0 ? 1 : 0) + (idxBooking >= 0 ? 1 : 0); if (!best || score > best.score) best = { row: r + 1, headers, score }; if (score === 3) break; } if (!best) return headerResult_(1, (values[0] || []).map(h => cleanText_(h))); return headerResult_(best.row, best.headers); } function buildDebug_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const allSheets = ss.getSheets().map(s => s.getName()); const sh = ss.getSheetByName(TAB_NAME); if (!sh) return { ok: false, error: "找不到工作表分頁(tab)", tabNameSetting: TAB_NAME, allSheets }; const lastRow = sh.getLastRow(); const lastCol = sh.getLastColumn(); const scanRows = Math.max(Math.min(HEADER_SCAN_MAX_ROWS, lastRow), 1); const allValues = sh.getRange(1, 1, scanRows, lastCol).getDisplayValues(); const h = resolveHeader_(allValues); const preview = allValues.map((row, i) => ({ row: i + 1, cells: row.map(c => cleanText_(c)) })); return { ok: true, tabNameSetting: TAB_NAME, allSheets, lastRow, lastCol, detectedHeaderRow: h.headerRowIndex, detectedHeaders: h.headers, dateColumnFoundAt: h.idxDate + 1, nameColumnFoundAt: h.idxName + 1, paidColumnFoundAt: h.idxPaid + 1, rescheduleColumnFoundAt: h.idxReschedule + 1, bookingColumnFoundAt: h.idxBooking + 1, intlColumnFoundAt: h.idxIntl + 1, firstRowsPreview: preview }; } function getRangeFromParams_(params) { const def = defaultRange_(); const s = String(params.start || "").trim(); const e = String(params.end || "").trim(); if (!s && !e) return def; let start = s ? parseYMD_(s) : null; let end = e ? parseYMD_(e) : null; if (start && !end) end = new Date(); if (!start && end) { start = new Date(end); start.setMonth(start.getMonth() - 6); } if (!start || !end) return def; start.setHours(0, 0, 0, 0); end.setHours(23, 59, 59, 999); if (start > end) throw new Error("日期區間錯誤：start 不能晚於 end"); return { start, end, label: "custom" }; } function parseYMD_(s) { const m = String(s).match(/^(\d{4})-(\d{2})-(\d{2})$/); if (!m) return null; const dt = new Date(Number(m[1]), Number(m[2]) - 1, Number(m[3])); return isNaN(dt) ? null : dt; } function ymd_(d) { const yyyy = d.getFullYear(); const mm = String(d.getMonth() + 1).padStart(2, "0"); const dd = String(d.getDate()).padStart(2, "0"); return yyyy + "-" + mm + "-" + dd; } function readRowsInRange_(range) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(TAB_NAME); if (!sh) throw new Error("找不到工作表分頁(tab)：" + TAB_NAME); const lastRow = sh.getLastRow(); const lastCol = sh.getLastColumn(); if (lastRow < 1) return []; const values = sh.getRange(1, 1, lastRow, lastCol).getDisplayValues(); const h = resolveHeader_(values); const headerRowIndex = h.headerRowIndex; const idxDate = h.idxDate, idxName = h.idxName, idxPaid = h.idxPaid; const idxReschedule = h.idxReschedule, idxBooking = h.idxBooking, idxIntl = h.idxIntl; if (idxDate < 0) throw new Error("找不到欄位：日期。讀到的欄位是：" + h.headers.join("｜")); if (idxPaid < 0) throw new Error("找不到欄位：已付款。讀到的欄位是：" + h.headers.join("｜")); if (idxBooking < 0) throw new Error("找不到欄位：預約日期。讀到的欄位是：" + h.headers.join("｜")); const rows = []; for (let r = headerRowIndex; r < values.length; r++) { const row = values[r]; const dateCell = row[idxDate]; const paidCell = row[idxPaid]; const bookingCell = row[idxBooking]; const nameCell = idxName >= 0 ? row[idxName] : ""; const intlCell = idxIntl >= 0 ? row[idxIntl] : ""; const rescheduleCell = idxReschedule >= 0 ? row[idxReschedule] : ""; if (isBlank_(dateCell) && isBlank_(paidCell) && isBlank_(bookingCell)) continue; const dt = parseDate_(dateCell); if (!dt) continue; if (dt < range.start || dt > range.end) continue; rows.push({ dateObj: dt, dateText: formatDate_(dt), name: cleanText_(nameCell), paid: isPaid_(paidCell), reschedule: isChecked_(rescheduleCell), booking: bookingCell, intl: isChecked_(intlCell) }); } rows.sort((a, b) => b.dateObj - a.dateObj); return rows; } // 統一的狀態判斷（順序：改期 → 退費 → 未預約 → 已預約） function classify_(r) { const bookingText = cleanText_(r.booking); if (r.reschedule) return "改期"; if (bookingText && hasAny_(bookingText, REFUND_KEYWORDS)) return "退費"; if (!bookingText) return "未預約"; return "已預約"; } function buildSummary_(range) { const rows = readRowsInRange_(range); let paidTotal = 0, paidNotBooked = 0, paidRefund = 0, paidRescheduled = 0, paidBooked = 0; let intlChecked = 0, intlNotBooked = 0; rows.forEach(r => { const label = classify_(r); if (r.intl) { intlChecked++; if (label === "未預約") intlNotBooked++; } if (!r.paid) return; paidTotal++; if (label === "未預約") paidNotBooked++; else if (label === "退費") paidRefund++; else if (label === "改期") paidRescheduled++; else paidBooked++; }); return { ok: true, range: range.label, start: ymd_(range.start), end: ymd_(range.end), paidTotal, paidNotBooked, paidRefund, paidRescheduled, paidBooked, intlChecked, intlNotBooked, updatedAt: new Date().toISOString() }; } function buildList_(status, range) { const all = readRowsInRange_(range); const isIntlStatus = (status === "intl_all" || status === "intl_not_booked"); const source = isIntlStatus ? all.filter(r => r.intl) : all.filter(r => r.paid); const items = []; source.forEach(r => { const label = classify_(r); let shouldPush = false; if (status === "paid_all" || status === "intl_all") shouldPush = true; else if (status === "not_booked") shouldPush = (label === "未預約"); else if (status === "refund") shouldPush = (label === "退費"); else if (status === "rescheduled") shouldPush = (label === "改期"); else if (status === "booked") shouldPush = (label === "已預約"); else if (status === "intl_not_booked") shouldPush = (label === "未預約"); if (shouldPush) { items.push({ "日期": r.dateText, "客戶本名": r.name, "預約日期": cleanText_(r.booking), "狀態": label }); } }); return { ok: true, status, range: range.label, start: ymd_(range.start), end: ymd_(range.end), count: items.length, items, updatedAt: new Date().toISOString() }; } function findHeaderIndex_(headers, candidates) { const cleanHeaders = headers.map(h => normalizeHeader_(h)); for (const candidate of candidates) { const idx = cleanHeaders.indexOf(normalizeHeader_(candidate)); if (idx >= 0) return idx; } return -1; } function cleanText_(v) { return String(v || "").replace(/\u3000/g, " ").replace(/\r?\n|\r/g, " ").trim(); } function normalizeHeader_(v) { return String(v || "").replace(/\u3000/g, "").replace(/\s+/g, "").trim(); } function isPaid_(v) { const s = cleanText_(v).toLowerCase(); if (!s) return false; return ["v", "✓", "✔", "true", "1", "yes", "y", "已付款", "已付", "付款", "已對帳"].includes(s); } function isChecked_(v) { const s = cleanText_(v).toLowerCase(); if (!s) return false; return ["v", "✓", "✔", "true", "1", "yes", "y", "是", "✅", "國際美金", "美金"].includes(s); } function hasAny_(text, keywords) { const t = cleanText_(text); return keywords.some(k => t.indexOf(k) >= 0); } function isBlank_(v) { return cleanText_(v) === ""; } function parseDate_(v) { const s = cleanText_(v); if (!s) return null; let m; m = s.match(/^(\d{4})[\/\-](\d{1,2})[\/\-](\d{1,2})/); if (m) { const dt = new Date(Number(m[1]), Number(m[2]) - 1, Number(m[3])); return isNaN(dt) ? null : dt; } m = s.match(/^(\d{2})[\/\-](\d{1,2})[\/\-](\d{1,2})/); if (m) { const dt = new Date(2000 + Number(m[1]), Number(m[2]) - 1, Number(m[3])); return isNaN(dt) ? null : dt; } return null; } function formatDate_(d) { const yyyy = d.getFullYear(); const mm = String(d.getMonth() + 1).padStart(2, "0"); const dd = String(d.getDate()).padStart(2, "0"); return yyyy + "/" + mm + "/" + dd;
